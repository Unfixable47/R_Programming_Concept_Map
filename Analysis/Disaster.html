  rename(Number_of_Disasters_Declared = number_of_incidents,Year_Declared = fy_declared)

  arrange(n)

#filter out outliers #outliers filters where the year 2020(for covid-19), and 2023(because it was incomplete collection of information)

  filter(Number_of_Disasters_Declared < 4500,Year_Declared < 2023)

boxplot(res$Number_of_Disasters_Declared)

#build an arrangment paired with mutaute in DPLYR library than pass it to ggplot and build the graph #this reorders the factors in the graph so its easier to break down and look at

  arrange(n) %>% 
  mutate(incident_type=factor(incident_type,levels=incident_type)) %>% 
  ggplot(aes(x=incident_type,y=n))+
